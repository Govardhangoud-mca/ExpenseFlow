import api from "../api";

interface AuthUser {
  email: string;
  role: "admin" | "developer" | "tester";
  name?: string;
}

interface AuthContextType {
  user: AuthUser | null;
  setUser: (user: AuthUser | null) => void; // Add setUser
  login: (email: string, password: string) => Promise<AuthUser>;
  logout: () => void;
  loading: boolean;
}

const AuthContext = createContext<AuthContextType | undefined>(undefined);

export const AuthProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [user, setUser] = useState<AuthUser | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    try {
      const storedUser = localStorage.getItem("user");
      if (storedUser && storedUser !== "undefined") {
        setUser(JSON.parse(storedUser));
      }
    } catch (error) {
      console.error("Error parsing user data from localStorage:", error);
      localStorage.removeItem("user");
    } finally {
      setLoading(false);
    }
  }, []);

  const login = async (email: string, password: string) => {
    try {
      const { data } = await api.post("api/auth/login", { email, password }, { withCredentials: true });

      if (!data || !data.user) throw new Error("Invalid email or password");

      const authUser: AuthUser = {
        email: data.user.email,
        role: data.user.role.toLowerCase() as AuthUser["role"],
        name: data.user.name,
      };

      // Save user immediately for navigation & ProtectedRoute
      setUser(authUser);
      localStorage.setItem("user", JSON.stringify(authUser));

      return authUser;
    } catch (error: any) {
      if (error.response && error.response.status === 401) {
        throw new Error(error.response.data.message || "Invalid email or password");
      }
      throw new Error(error.message || "Login failed");
    }
  };

  const logout = () => {
    setUser(null);
    localStorage.removeItem("user");
    localStorage.removeItem("token");
  };

  return (
    <AuthContext.Provider value={{ user, setUser, login, logout, loading }}>
      {children}
    </AuthContext.Provider>
  );
};

export const useAuth = () => {
  const context = useContext(AuthContext);
  if (!context) throw new Error("useAuth must be used within an AuthProvider");
  return context;
};

import { Bug, Project, BugFormData } from '@/types/Bug';
import { bugService } from '@/services/bugService';

interface BugContextType {
  bugs: Bug[];
  projects: Project[];
  loading: boolean;
  createBug: (data: BugFormData) => Promise<Bug>;
  updateBug: (id: string, data: Partial<Bug>) => Promise<Bug>;
  deleteBug: (id: string) => Promise<void>;
  getBugById: (id: string) => Bug | undefined;
  refreshBugs: () => Promise<void>;
}

const BugContext = createContext<BugContextType | undefined>(undefined);

export function BugProvider({ children }: { children: React.ReactNode }) {
  const [bugs, setBugs] = useState<Bug[]>([]);
  const [projects, setProjects] = useState<Project[]>([]);
  const [loading, setLoading] = useState(true);

  const refreshBugs = async () => {
    setLoading(true);
    try {
      const [bugsResponse, projectsResponse] = await Promise.all([
        bugService.getAllBugs(),
        bugService.getAllProjects(),
      ]);
      setBugs(bugsResponse.data);
      setProjects(projectsResponse.data);
    } catch (error) {
      console.error("Failed to fetch bugs or projects:", error);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    refreshBugs();
  }, []);

  const createBug = async (data: BugFormData): Promise<Bug> => {
    const res = await bugService.createBug(data);
    setBugs(prev => [res.data, ...prev]);
    return res.data;
  };

  const updateBug = async (id: string, data: Partial<Bug>): Promise<Bug> => {
    const res = await bugService.updateBug(id, data);
    setBugs(prev => prev.map(bug => bug.id === id ? res.data : bug));
    return res.data;
  };

  const deleteBug = async (id: string): Promise<void> => {
    await bugService.deleteBug(id);
    setBugs(prev => prev.filter(bug => bug.id !== id));
  };

  const getBugById = (id: string): Bug | undefined => bugs.find(bug => bug.id === id);

  return (
    <BugContext.Provider value={{ bugs, projects, loading, createBug, updateBug, deleteBug, getBugById, refreshBugs }}>
      {children}
    </BugContext.Provider>
  );
}

export function useBugs() {
  const context = useContext(BugContext);
  if (!context) throw new Error('useBugs must be used within a BugProvider');
  return context;
}

export const UserProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [users, setUsers] = useState<User[]>([]);
  const [loading, setLoading] = useState(false);

  const fetchUsers = async () => {
    setLoading(true);
    try {
      const res = await api.get("/users"); // session cookie sent automatically
      setUsers(res.data);
    } catch (error) {
      console.error(error);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchUsers();
  }, []);

  return (
    <UserContext.Provider value={{ users, loading, fetchUsers }}>
      {children}
    </UserContext.Provider>
  );
};

export const useUsers = () => {
  const context = useContext(UserContext);
  if (!context) throw new Error("useUsers must be used within UserProvider");
  return context;
};
